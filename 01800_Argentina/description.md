     size 4 3
     cell 0 0 Azul 1
     cell 1 0 Azul 1
     cell 3 0 Azul 1
     cell 4 0 Azul 1
     cell 0 3 Azul 1
     cell 1 3 Azul 1
     cell 3 3 Azul 1
     cell 4 3 Azul 1
     head 4 3
<gs-board>
